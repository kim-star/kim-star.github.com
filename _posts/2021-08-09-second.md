'''python
#befor

fruits = ['lemon','kiwi','orange']
index = 0
kiwi = []

for i, fruit in enumerate(fruits) :
  if (fruit=='kiwi') :
    index=i
    kiwi=fruit    #shallow copykiwi와 fruit가 같은 곳을 가리키고 있어. 치명적인 오류.

print(index, kiwi)

#after

import copy

fruits = ['lemon','kiwi','orange']
index = 0
kiwi = []

for i, fruit in enumerate(fruits) :
  if (fruit=='kiwi') :
    index=i
    kiwi=copy.deepcopy(fruit)   #copy.deepcopy()
    
print(index, kiwi)
